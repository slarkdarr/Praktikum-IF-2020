jumlah(1,X,Y) :- suksesor(X,Y).
jumlah(X1,Y,Z1) :- 
    suksesor(X,X1),
    jumlah(X,Y,Z),
    suksesor(Z,Z1).

jumlah(3,6,Z) :-
    suksesor(X,3),  // X = 2
    jumlah(2,6,Z),
        suksesor(X,2),  // X = 1
        jumlah(1,6,Z),  // Z = 7
        suksesor(7,Z1). // Z1 = 8
    suksesor(8,Z1). // Z1 = 9

anak(X)
